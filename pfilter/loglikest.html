Now, we evaluate the dependence of log likelihood estimates on particle size and number of independent filters. We first set up a parallel computation environment.

library(foreach)
library(doParallel)
registerDoParallel()
expand_grid(
  rep=1:1000,
  Np=c(1000,10000,100000)
) -> design

foreach (p=iter(design,"row"),
         .inorder=FALSE, .combine=rbind) %dopar%
  {
    library(pomp)
    measSIR %>% pfilter(Np=p$Np) -> pf
    cbind(p,loglik=logLik(pf))
  } -> lls

Violin plots are cute.

expand_grid(
  nfilt=c(10,100,1000),
  lls
) %>%
  filter(rep<=nfilt) %>%
  ggplot(aes(x=Np,y=loglik,fill=ordered(nfilt),
             group=interaction(nfilt,Np)))+
  geom_violin(draw_quantiles=c(0.1,0.5,0.9),alpha=0.7)+
  scale_x_log10(breaks=unique(lls$Np))+
  labs(fill="nfilt")
